    module RailsStars::Helper

    The RailsStars::Helper mixes in ActionView::Base to give the #stars_for method that displays stars ratings and let visitors click on to give their own rating.

def display_stars receiver
  ratings = rounded_star_average(receiver)
  render :partial => "rateable/rating", :locals => {:stars => 5, :rating => ratings.to_i, :tooltip => " "}
end

def stars_form_for receiver
  form_for([rails_stars, RailsStars::Star.new(anchor: receiver.stars_anchor!)], html: {
           :"data-stars-average" => rounded_star_average(receiver),
           :"data-stars-count" => receiver.star_count,
           :id => nil}) do |f|
    [
      f.hidden_field(:anchor_id),
      f.hidden_field(:rating)
    ].join.html_safe
  end
end
